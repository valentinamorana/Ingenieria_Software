  1. DIAGRAMA DE SECUENCIA — LOGIN (éxito + fallo + bloqueo)
========================================================

@startuml DS_Login
title Diagrama de Secuencia — Login

skinparam sequenceArrowThickness 1.5
skinparam backgroundColor #FFF0F3
skinparam sequenceParticipantBackgroundColor #FCE4EB
skinparam sequenceParticipantBorderColor #D2638A

actor       "Empleado"          as U
participant "GUI.Login"         as GL
participant "BLL.Usuario"       as BU
participant "DAL.Usuario"       as DU
participant "Encriptador"       as E
participant "DAL.Permiso"       as DP
participant "SessionManager"    as SM
participant "Servicios.Bitacora" as SB
participant "DAL.Bitacora"      as DB

U  -> GL : ingresa username + contraseña\npresiona Ingresar / Enter

GL -> BU : Login(form, username, contraseña)

BU -> BU : validar campos no vacíos\n[lanza excepción si vacíos]

BU -> DU : ObtenerPorUsername(username)
DU --> BU : BE.Usuario (o null)

alt usuario == null
    BU --> GL : return false
    GL --> U  : sin mensaje (usuario no existe)
end

BU -> BU : usuario.Bloqueado ?

alt cuenta bloqueada
    BU --> GL : throw "Cuenta bloqueada.\nContactar Administrador"
    GL --> U  : lblError = mensaje
end

BU -> E : VerificarContraseña(contraseña, usuario.Contraseña)
note right of E : PBKDF2-SHA256\nExtrae salt → rehashea\n100.000 iteraciones\ncompara byte a byte
E --> BU : bool esValido

alt esValido == true
    BU -> DU : ResetearIntentosFallidos(username)
    BU -> DP : ObtenerPorRol(usuario.Rol)
    DP --> BU : List<BE.Permiso>
    BU -> BU : usuario.Permisos = permisos
    BU -> SM : Login(usuario)
    note right of SM : Crea instancia Singleton\n_session = new SessionManager()\n_session.Usuario = usuario
    SM --> BU : ok
    BU -> SB : Registrar(form.Text, "Inicio Sesion", None)
    SB -> SM : GetInstance (obtiene usuario en sesión)
    SB -> DB : Registrar(BE.Bitacora)
    DB --> SB : ok
    BU --> GL : return true
    GL -> GL : DialogResult = OK\nClose()
    GL --> U  : formulario se cierra\nse abre GUI.Menu

else esValido == false
    BU -> DU : IncrementarIntentosFallidos(username)
    BU -> BU : intentos = usuario.IntentosFallidos + 1

    BU -> SB : RegistrarSinSesion("Login",\n"Intento Fallido Login", IntentosLogin,\nidUsuario, detalle)
    SB -> DB : Registrar(BE.Bitacora) [sin sesión]
    DB --> SB : ok

    alt intentos >= 3
        BU -> DU : Bloquear(usuario.Id)
        BU -> SB : RegistrarSinSesion("Login",\n"Bloqueo de Cuenta", BloqueosCuenta,\nidUsuario, detalle)
        SB -> DB : Registrar(BE.Bitacora)
        BU --> GL : throw "Cuenta bloqueada tras 3 intentos"
        GL --> U  : lblError = mensaje de bloqueo
    else intentos < 3
        BU --> GL : throw "Credenciales incorrectas.\nIntentos restantes: N"
        GL --> U  : lblError = mensaje\ntxtContraseña.Clear()
    end
end
@enduml

U  -> GM : clic "Perfil" → "Cerrar Sesión"
GM -> GM : MessageBox.Show\n"¿Está seguro?"

alt usuario confirma Yes
    GM -> BU : Logout(this)

    BU -> SB : Registrar(form.Text, "Cierre Sesion", None)
    SB -> SM : GetInstance (lee usuario en sesión)
    SM --> SB : SessionManager con Usuario
    SB -> SB : construye BE.Bitacora con\nIP, fecha, detalle
    SB -> DB : Registrar(BE.Bitacora)
    DB --> SB : ok

    BU -> DU : Logout() [registra fecha último logout en BD]
    DU --> BU : ok

    BU -> SM : Logout()
    note right of SM : _session = null\nGC libera el objeto\nSingleton destruido
    SM --> BU : ok

    BU --> GM : retorna
    GM -> GM : Application.Restart()
    GM --> U  : app reinicia → aparece GUI.Login

else usuario cancela No
    GM --> U  : cierra el diálogo\npermanece en el sistema
end
@enduml

  3. DIAGRAMA DE SECUENCIA — BITÁCORA (apertura + filtro)
========================================================

@startuml DS_Bitacora
title Diagrama de Secuencia — Bitácora (apertura y filtrado)

skinparam sequenceArrowThickness 1.5
skinparam backgroundColor #FFF0F3
skinparam sequenceParticipantBackgroundColor #FCE4EB
skinparam sequenceParticipantBorderColor #D2638A

actor       "Administrador / Supervisor" as U
participant "GUI.Menu"                   as GM
participant "GUI.Bitacora"               as GB
participant "Servicios.Bitacora"         as SB
participant "Servicios.BitacoraNegocio"  as SBN
participant "DAL.Bitacora"               as DB
participant "DAL.BitacoraNegocio"        as DBN

== Apertura del módulo ==

U  -> GM : clic "Bitácora" en menú
GM -> GM : verifica que no haya instancia ya abierta
GM -> GB : new Bitacora { MdiParent = this }.Show()
GB -> GB : Bitacora_Load()

GB -> SB : ObtenerTodos()
SB -> DB : ObtenerTodos()\nSELECT * FROM Bitacora ORDER BY fecha DESC
DB --> SB : DataTable
SB --> GB : DataTable

GB -> SBN : ObtenerTodos()
SBN -> DBN : ObtenerTodos()\nSELECT * FROM BitacoraNegocio ORDER BY Fecha DESC
DBN --> SBN : DataTable
SBN --> GB : DataTable

GB -> GB : MostrarEnGrilla(dgvSistema, dt)\nMostrarEnGrilla(dgvNegocio, dt)
GB --> U  : muestra ambas pestañas con datos

== Filtro por últimos N días (pestaña Sistema) ==

U  -> GB : ingresa valor en nudDias\npresiona "Últimos N días"
GB -> GB : dias = nudDias.Value

alt dias == 0
    GB -> SB : ObtenerTodos()
    SB -> DB : ObtenerTodos()
else dias > 0
    GB -> SB : ObtenerUltimosNDias(dias)
    SB -> SB : normaliza: if dias < 1 → dias = 1
    SB -> DB : ObtenerUltimosNDias(dias)\nWHERE fecha >= NOW() - dias
end

DB --> SB : DataTable
SB --> GB : DataTable
GB -> GB : MostrarEnGrilla(dgvSistema, ...\n"últimos N días")
GB --> U  : grilla actualizada + lblResultados

== Filtro combinado (búsqueda avanzada) ==

U  -> GB : completa filtros:\nfecha desde/hasta, criticidad, actividad\npresiona "Buscar"

GB -> SB : BuscarPorFiltros(desde, hasta, idUsuario, actividad, criticidad)
SB -> DB : BuscarPorFiltros(...)\nWHERE 1=1 + cláusulas dinámicas
DB --> SB : DataTable
SB --> GB : DataTable
GB --> U  : grilla filtrada

== Exportar PDF ==

U  -> GB : clic "Exportar PDF"
GB -> GB : prepara _tablaImpresion\n_tituloImpresion = tab activo
GB -> GB : PrintPreviewDialog.ShowDialog()
GB --> U  : previsualizador\nUsuario imprime → "Microsoft Print to PDF"
@enduml
